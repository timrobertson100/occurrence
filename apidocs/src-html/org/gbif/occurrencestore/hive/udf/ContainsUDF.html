001package org.gbif.occurrencestore.hive.udf;
002
003import java.util.Map;
004
005import com.google.common.collect.Maps;
006import com.vividsolutions.jts.geom.Coordinate;
007import com.vividsolutions.jts.geom.Geometry;
008import com.vividsolutions.jts.geom.GeometryFactory;
009import com.vividsolutions.jts.geom.PrecisionModel;
010import com.vividsolutions.jts.io.ParseException;
011import com.vividsolutions.jts.io.WKTReader;
012import org.apache.hadoop.hive.ql.exec.Description;
013import org.apache.hadoop.hive.ql.exec.UDF;
014import org.apache.hadoop.io.BooleanWritable;
015import org.apache.hadoop.io.Text;
016
017/**
018 * A simple UDF for Hive to test if a coordinate is contained in a geometry given as well known text.
019 * Any geometry supported by jts is accepted, but its up to the user to make sensible queries.
020 * In particular polygons should have a complete exterior ring.
021 */
022@Description(name = "contains", value = "_FUNC_(geom_wkt, latitude, longitude)")
023public class ContainsUDF extends UDF {
024
025  private final WKTReader wktReader = new WKTReader();
026  private final GeometryFactory geometryFactory = new GeometryFactory(new PrecisionModel(), 4326);
027  private final Map<String, Geometry> geometryCache = Maps.newHashMap();
028  private final BooleanWritable isContained = new BooleanWritable();
029
030  public BooleanWritable evaluate(Text geometryAsWKT, Double latitude, Double longitude) {
031    isContained.set(false);
032
033    try {
034      // sanitize the input
035      if (geometryAsWKT == null || latitude == null || longitude == null || latitude > 90 || latitude < -90
036        || longitude > 180 || longitude < -180) {
037        return isContained;
038      }
039      String geoWKTStr = geometryAsWKT.toString();
040      Geometry geom = geometryCache.get(geoWKTStr);
041      if (geom == null) {
042        geom = wktReader.read(geoWKTStr);
043        geometryCache.put(geoWKTStr, geom);
044      }
045
046      // support any geometry - up to the user to make a sensible query
047      Geometry point = geometryFactory.createPoint(new Coordinate(longitude, latitude));
048      isContained.set(geom.contains(point));
049
050    } catch (ParseException e) {
051      System.err.println("Invalid geometry received: " + geometryAsWKT.toString());
052    } catch (Exception e) {
053      e.printStackTrace();
054    }
055
056    return isContained;
057  }
058}
