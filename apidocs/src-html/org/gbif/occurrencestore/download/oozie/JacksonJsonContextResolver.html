001package org.gbif.occurrencestore.download.oozie;
002
003import java.util.Map;
004
005import javax.ws.rs.Produces;
006import javax.ws.rs.core.MediaType;
007import javax.ws.rs.ext.ContextResolver;
008import javax.ws.rs.ext.Provider;
009
010import com.google.inject.Singleton;
011import org.codehaus.jackson.map.DeserializationConfig;
012import org.codehaus.jackson.map.ObjectMapper;
013import org.codehaus.jackson.map.SerializationConfig;
014import org.codehaus.jackson.map.annotate.JsonSerialize;
015
016/**
017 * Provider that initializes the {@link ObjectMapper} to ignore {@code null} fields and unknown properties.
018 * Copied and modified from gbif-common-ws to be compatible with jackson 1.8.8 used in the cdh4 environment.
019 */
020@Provider
021@Produces(MediaType.APPLICATION_JSON)
022@Singleton
023public class JacksonJsonContextResolver implements ContextResolver<ObjectMapper> {
024
025  private static final ObjectMapper MAPPER = new ObjectMapper();
026
027  static {
028    // determines whether encountering of unknown properties (ones that do not map to a property, and there is no
029    // "any setter" or handler that can handle it) should result in a failure (throwing a JsonMappingException) or not.
030    MAPPER.configure(DeserializationConfig.Feature.FAIL_ON_UNKNOWN_PROPERTIES, false);
031    MAPPER.configure(SerializationConfig.Feature.WRITE_DATES_AS_TIMESTAMPS, false);
032    MAPPER.getSerializationConfig().setSerializationInclusion(JsonSerialize.Inclusion.NON_NULL);
033  }
034
035  /**
036   * Specific polymorphic serialization is supported by passing a map of mixIns into the class.
037   *
038   * @param mixIns class map
039   */
040  public static void addMixIns(Map<Class<?>, Class<?>> mixIns) {
041    // handle polymorphic JSON
042    for (Map.Entry<Class<?>, Class<?>> classClassEntry : mixIns.entrySet()) {
043      MAPPER.getSerializationConfig().addMixInAnnotations(classClassEntry.getKey(), classClassEntry.getValue());
044      MAPPER.getDeserializationConfig().addMixInAnnotations(classClassEntry.getKey(), classClassEntry.getValue());
045    }
046  }
047
048  @Override
049  public ObjectMapper getContext(Class<?> type) {
050    return MAPPER;
051  }
052}
