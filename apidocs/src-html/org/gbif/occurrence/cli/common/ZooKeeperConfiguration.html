001package org.gbif.occurrence.cli.common;
002
003import java.io.IOException;
004import javax.validation.constraints.Min;
005import javax.validation.constraints.NotNull;
006
007import com.beust.jcommander.Parameter;
008import com.fasterxml.jackson.annotation.JsonIgnore;
009import com.google.common.base.Objects;
010import com.netflix.curator.framework.CuratorFramework;
011import com.netflix.curator.framework.CuratorFrameworkFactory;
012import com.netflix.curator.retry.ExponentialBackoffRetry;
013
014/**
015 * A configuration class which can be used to get all the details needed to create a connection to ZooKeeper needed by
016 * the Curator Framework.
017 * It provides a convenience method ({@link #getCuratorFramework()} ()} to actually get a {@link
018 * com.netflix.curator.framework.CuratorFramework}
019 * object when populated fully.
020 */
021@SuppressWarnings("PublicField")
022public class ZooKeeperConfiguration {
023
024  @Parameter(
025    names = "--zk-connection-string",
026    description = "The connection string to connect to ZooKeeper")
027  @NotNull
028  public String connectionString;
029
030  @Parameter(
031    names = "--zk-namespace",
032    description = "The namespace in ZooKeeper under which all data lives")
033  @NotNull
034  public String namespace;
035
036  @Parameter(
037    names = "--zk-sleep-time",
038    description = "Initial amount of time to wait between retries in ms")
039  @Min(1)
040  public int baseSleepTime = 1000;
041
042  @Parameter(
043    names = "--zk-max-retries",
044    description = "Max number of times to retry")
045  @Min(1)
046  public int maxRetries = 10;
047
048  /**
049   * This method returns a connection object to ZooKeeper with the provided settings and creates and starts a {@link
050   * com.netflix.curator.framework.CuratorFramework}. These settings are not validated in this method so only call it
051   * when the object has been
052   * validated.
053   *
054   * @return started CuratorFramework
055   *
056   * @throws java.io.IOException if connection fails
057   */
058  @JsonIgnore
059  public CuratorFramework getCuratorFramework() throws IOException {
060    CuratorFramework curator = CuratorFrameworkFactory.builder().namespace(namespace)
061      .retryPolicy(new ExponentialBackoffRetry(baseSleepTime, maxRetries)).connectString(connectionString).build();
062    curator.start();
063    return curator;
064  }
065
066  @Override
067  public String toString() {
068    return Objects.toStringHelper(this).add("connectionString", connectionString).add("namespace", namespace)
069      .add("baseSleepTime", baseSleepTime).add("maxRetries", maxRetries).toString();
070
071  }
072}
