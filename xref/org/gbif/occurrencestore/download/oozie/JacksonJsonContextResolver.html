16  /**
17   * Provider that initializes the {@link ObjectMapper} to ignore {@code null} fields and unknown properties.
18   * Copied and modified from gbif-common-ws to be compatible with jackson 1.8.8 used in the cdh4 environment.
19   */
20  @Provider
21  @Produces(MediaType.APPLICATION_JSON)
22  @Singleton
23  public class JacksonJsonContextResolver implements ContextResolver<ObjectMapper> {
24  
25    private static final ObjectMapper MAPPER = new ObjectMapper();
26  
27    static {
28      // determines whether encountering of unknown properties (ones that do not map to a property, and there is no
29      // "any setter" or handler that can handle it) should result in a failure (throwing a JsonMappingException) or not.
30      MAPPER.configure(DeserializationConfig.Feature.FAIL_ON_UNKNOWN_PROPERTIES, false);
31      MAPPER.configure(SerializationConfig.Feature.WRITE_DATES_AS_TIMESTAMPS, false);
32      MAPPER.getSerializationConfig().setSerializationInclusion(JsonSerialize.Inclusion.NON_NULL);
33    }
34  
35    /**
36     * Specific polymorphic serialization is supported by passing a map of mixIns into the class.
37     *
38     * @param mixIns class map
39     */
40    public static void addMixIns(Map<Class<?>, Class<?>> mixIns) {
41      // handle polymorphic JSON
42      for (Map.Entry<Class<?>, Class<?>> classClassEntry : mixIns.entrySet()) {
43        MAPPER.getSerializationConfig().addMixInAnnotations(classClassEntry.getKey(), classClassEntry.getValue());
44        MAPPER.getDeserializationConfig().addMixInAnnotations(classClassEntry.getKey(), classClassEntry.getValue());
45      }
46    }
47  
48    @Override
49    public ObjectMapper getContext(Class<?> type) {
50      return MAPPER;
51    }
