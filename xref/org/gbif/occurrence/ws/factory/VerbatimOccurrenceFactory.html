18  /**
19   * This is responsible for transforming the verbatim occurrence persistence object
20   * into the API representation which is Map based. All map entries are keyed using
21   * the DwC simple term names. Transposition of terms that don't map is implemented for
22   * <ul>
23   * <li>MinAltitude, MaxAltitude, Altitude Precision -> verbatimElevation</li>
24   * <li>MinDepth, MaxDepth, Depth Precision -> verbatimDepth</li>
25   * <li>yearIdentified, monthIdentified, dayIdentified, dateIdentified -> dateIdentified</li>
26   * </ul>
27   */

30    public static VerbatimOccurrence build(org.gbif.occurrencestore.persistence.api.VerbatimOccurrence from) {
31      VerbatimOccurrence vo = new VerbatimOccurrence();
32      vo.setKey(from.getKey());
33      Builder<Term, String> b = ImmutableMap.builder();
34      putIfNotNull(b, DwcTerm.scientificName, from.getScientificName());
35      putIfNotNull(b, DwcTerm.scientificNameAuthorship, from.getAuthor());
36      putIfNotNull(b, DwcTerm.taxonRank, from.getRank());
37      putIfNotNull(b, DwcTerm.kingdom, from.getKingdom());
38      putIfNotNull(b, DwcTerm.phylum, from.getPhylum());
39      putIfNotNull(b, DwcTerm.class_, from.getKlass());
40      putIfNotNull(b, DwcTerm.order, from.getOrder());
41      putIfNotNull(b, DwcTerm.family, from.getFamily());
42      putIfNotNull(b, DwcTerm.genus, from.getGenus());
43      putIfNotNull(b, DwcTerm.specificEpithet, from.getSpecies());
44      putIfNotNull(b, DwcTerm.infraspecificEpithet, from.getSubspecies());
45      putIfNotNull(b, DwcTerm.verbatimLatitude, from.getLatitude());
46      putIfNotNull(b, DwcTerm.verbatimLongitude, from.getLongitude());
47      // based on the belief that we most commonly mapped uncertaintity in the OLD portal
48      putIfNotNull(b, DwcTerm.coordinateUncertaintyInMeters, from.getLatLongPrecision());
49      putIfNotNull(b, DwcTerm.verbatimElevation,
50        minMaxPrecision(from.getMinAltitude(), from.getMaxAltitude(), from.getAltitudePrecision()));
51      putIfNotNull(b, DwcTerm.verbatimDepth, minMaxPrecision(from.getMinDepth(), from.getMaxDepth(), from.getDepthPrecision()));
52      // DwC has expanded continentOrOcean into 2 terms, so they may be collapsed
53      putIfNotNull(b, DwcTerm.continent, from.getContinentOrOcean(), "(originally supplied in continent OR ocean)");
54      putIfNotNull(b, DwcTerm.country, from.getCountry());
55      putIfNotNull(b, DwcTerm.stateProvince, from.getStateOrProvince());
56      putIfNotNull(b, DwcTerm.county, from.getCounty());
57      putIfNotNull(b, DwcTerm.recordedBy, from.getCollectorName());
58      putIfNotNull(b, DwcTerm.locality, from.getLocality());
59      putIfNotNull(b, DwcTerm.year, from.getYear());
60      putIfNotNull(b, DwcTerm.month, from.getMonth());
61      putIfNotNull(b, DwcTerm.day, from.getDay());
62      putIfNotNull(b, DwcTerm.eventDate, from.getOccurrenceDate());
63      putIfNotNull(b, DwcTerm.basisOfRecord, from.getBasisOfRecord());
64      putIfNotNull(b, DwcTerm.identifiedBy, from.getIdentifierName());
65      putIfNotNull(b, DwcTerm.dateIdentified,
66        yearMonthDayDate(from.getYearIdentified(), from.getMonthIdentified(), from.getDayIdentified(), from.getDateIdentified()));
67      vo.setFields(b.build());
68      return vo;
69    }

71    private static String minMaxPrecision(String min, String max, String precision) {
72      List<String> s = Lists.newArrayList();
73      if (min != null) {
74        s.add("Min[" + min + "]");
75      }
76      if (max != null) {
77        s.add("Max[" + max + "]");
78      }
79      if (precision != null) {
80        s.add("Precision[" + precision + "]");
81      }
82  
83      if (!s.isEmpty()) {
84        return Joiner.on(" ").join(s);
85      }
86      return null;
87    }

90    private static void putIfNotNull(Builder<Term, String> b, Term k, @Nullable String v) {
91      if (v != null) {
92        b.put(k, v);
93      }
94    }

96    private static void putIfNotNull(Builder<Term, String> b, Term k, @Nullable String v, String suffix) {
97      if (v != null) {
98        putIfNotNull(b, k, v + ' ' + suffix);
99      }
100   }

102   private static String yearMonthDayDate(String year, String month, String day, String date) {
103     List<String> s = Lists.newArrayList();
104     if (year != null) {
105       s.add("Year[" + year + ']');
106     }
107     if (month != null) {
108       s.add("Month[" + month + ']');
109     }
110     if (day != null) {
111       s.add("Day[" + day + ']');
112     }
113     if (date != null) {
114       s.add("Date[" + date + ']');
115     }
116 
117     if (!s.isEmpty()) {
118       return Joiner.on(" ").join(s);
119     }
120     return null;
121   }
